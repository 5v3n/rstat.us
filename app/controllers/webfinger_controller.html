  def xrd
    @user = Webfinger.find_user(params[:username])

    if @user
      @base_url = root_url
      render "xml/webfinger/xrd", :layout => false, :content_type => "application/xrd+xml"
    else
      render :file => "#{Rails.root}/public/404.html", :status => 404
    end
  end
