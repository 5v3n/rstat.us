  def avatar
    h.content_tag "div", :class => "avatar" do
      if author
        h.link_to(
          avatar_image_tag,

      else
        avatar_image_tag
      end
    end
  end

  def absolute_website_url
    if author.website.blank?
      nil
    elsif author.website[0,7] == "http://" or author.website[0,8] == "https://"
      author.website
    else
      "http://#{author.website}"
    end
  end
